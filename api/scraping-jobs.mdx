## Overview

The Scraping Jobs API provides helper functions for interacting with the multi-tenant scraping jobs system. These utilities manage influencer scraping configurations, job statistics, usernames, and assignments.

## Import

```typescript
import {
  createScrapingJob,
  getAllJobs,
  getJobById,
  getJobsByPlatform,
  getJobsByStatus,
  getActiveJob,
  getActiveBaseId,
  updateJob,
  updateJobStatus,
  getJobStatisticsByBaseId,
  getJobStatistics,
  getJobWithStats,
  getAllJobsWithStats,
  getJobUsernames,
  getJobAssignments,
  addUsernamesToJob,
  createJobAssignments,
  archiveJob,
  pauseJob,
  activateJob,
  searchJobsByInfluencer
} from '@/lib/scraping-jobs';
```

## Job Creation & Retrieval

### createScrapingJob

Create a new scraping job for an influencer.

```typescript
async function createScrapingJob(
  input: CreateScrapingJobInput
): Promise<ScrapingJob | null>
```

<ParamField path="input" type="CreateScrapingJobInput" required>
  Configuration for the new scraping job
  
  <ParamField path="input.influencer_name" type="string" required>
    Name of the influencer
  </ParamField>
  
  <ParamField path="input.platform" type="Platform" required>
    Social media platform: `'instagram'` | `'threads'` | `'tiktok'` | `'x'`
  </ParamField>
  
  <ParamField path="input.airtable_base_id" type="string" required>
    Airtable base ID for multi-tenant isolation
  </ParamField>
  
  <ParamField path="input.num_vas" type="number">
    Number of virtual assistants (optional)
  </ParamField>
  
  <ParamField path="input.status" type="JobStatus">
    Initial job status: `'active'` | `'paused'` | `'archived'` (defaults to `'active'`)
  </ParamField>
</ParamField>

<ResponseField name="ScrapingJob" type="ScrapingJob | null">
  Returns the created job object or `null` if creation failed
  
  <ResponseField name="job_id" type="string">
    Unique job identifier (UUID)
  </ResponseField>
  
  <ResponseField name="influencer_name" type="string">
    Name of the influencer
  </ResponseField>
  
  <ResponseField name="platform" type="Platform">
    Social media platform
  </ResponseField>
  
  <ResponseField name="airtable_base_id" type="string">
    Airtable base ID
  </ResponseField>
  
  <ResponseField name="base_id" type="string">
    Multi-tenant isolation key (source of truth from DB)
  </ResponseField>
  
  <ResponseField name="num_vas" type="number | null">
    Number of virtual assistants
  </ResponseField>
  
  <ResponseField name="status" type="JobStatus">
    Current job status
  </ResponseField>
  
  <ResponseField name="created_at" type="string">
    ISO 8601 timestamp of creation
  </ResponseField>
  
  <ResponseField name="updated_at" type="string | null">
    ISO 8601 timestamp of last update
  </ResponseField>
</ResponseField>

**Example:**

```typescript
const job = await createScrapingJob({
  influencer_name: 'John Doe',
  platform: 'instagram',
  airtable_base_id: 'appXXXXXXXXXXXXXX',
  num_vas: 5,
  status: 'active'
});

if (job) {
  console.log('Created job:', job.job_id);
}
```

### getActiveJob

Get the active scraping job. Returns the first job with `status='active'`, or the most recent job if none are active.

This is the primary method for determining which `base_id` to use for multi-tenant operations.

```typescript
async function getActiveJob(): Promise<ScrapingJob | null>
```

<ResponseField name="job" type="ScrapingJob | null">
  The active job, or `null` if no jobs exist
</ResponseField>

**Example:**

```typescript
const activeJob = await getActiveJob();
if (activeJob) {
  console.log('Active job base_id:', activeJob.base_id);
}
```

async function getActiveBaseId(): Promise<string | null>

## Job Statistics

### getJobStatisticsByBaseId

Get job statistics by `base_id`. Queries `global_usernames` and `daily_assignments` using `base_id` for accurate multi-tenant counts.

**This is the recommended method for getting statistics.**

```typescript
async function getJobStatisticsByBaseId(
  baseId: string
): Promise<JobStatistics | null>
```

<ParamField path="baseId" type="string" required>
  The base ID to query statistics for
</ParamField>

<ResponseField name="statistics" type="JobStatistics | null">
  Job statistics or `null` if query failed
  
  <ResponseField name="totalUsernames" type="number">
    Total number of usernames in the global pool
  </ResponseField>
  
  <ResponseField name="usedUsernames" type="number">
    Number of usernames marked as used
  </ResponseField>
  
  <ResponseField name="availableUsernames" type="number">
    Number of available (unused) usernames
  </ResponseField>
  
  <ResponseField name="totalAssignments" type="number">
    Total number of daily assignments
  </ResponseField>
  
  <ResponseField name="totalScrapeResults" type="number">
    Total number of scrape results
  </ResponseField>
</ResponseField>

**Example:**

```typescript
const stats = await getJobStatisticsByBaseId('appXXXXXXXXXXXXXX');
if (stats) {
  console.log(`${stats.availableUsernames} / ${stats.totalUsernames} usernames available`);
}
```

### addUsernamesToJob

Add usernames to a job's global username pool.

```typescript
async function addUsernamesToJob(
  jobId: string,
  baseId: string,
  usernames: Array<{
    id: string;
    username: string;
    full_name?: string;
  }>
): Promise<boolean>
```

<ParamField path="jobId" type="string" required>
  The job ID to add usernames to
</ParamField>

<ParamField path="baseId" type="string" required>
  The base_id for multi-tenant isolation
</ParamField>

<ParamField path="usernames" type="Array<object>" required>
  Array of username objects to add
  
  <ParamField path="usernames[].id" type="string" required>
    Unique user identifier
  </ParamField>
  
  <ParamField path="usernames[].username" type="string" required>
    The username
  </ParamField>
  
  <ParamField path="usernames[].full_name" type="string">
    User's full name (optional)
  </ParamField>
</ParamField>

<ResponseField name="success" type="boolean">
  `true` if usernames were added successfully
</ResponseField>

**Example:**

```typescript
const baseId = await getActiveBaseId();
if (baseId) {
  await addUsernamesToJob(jobId, baseId, [
    { id: '123', username: 'user1', full_name: 'User One' },
    { id: '456', username: 'user2', full_name: 'User Two' }
  ]);
}
```
